\subsection{Toy example}

\begin{frame}
\frametitle{Let's discuss}

List of film genres:
\begin{itemize}
	\item \midemp{sci-fi, fantastic, fantasy, cyber-punk, post-apo, documentary,
		drama, romance, \dots}
\end{itemize}

\vspace{1.2em}

We can \turemp{derive} genres from others, e.g:
\begin{itemize}
	\item \midemp{cyber-punk} leads probably to \midemp{sci-fi},
	\item a \midemp{documentary} may not be \midemp{fantastic},
	\item \dots
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Implications at first sight}

Rewritting of previous examples:
\begin{itemize}
	\item \midemp{cyber-punk} $\imp$ \midemp{sci-fi},
	\item \midemp{documentary, war} $\imp$ \midemp{historical},
\end{itemize}

\end{frame}
